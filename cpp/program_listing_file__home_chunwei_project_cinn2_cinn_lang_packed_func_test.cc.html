TEST(Function, ReturnMultiValue) {
  PackedFunc::body_t body = [](Args args, RetValue* ret) {
    int a = args[0];
    int b = args[1];
    int c = a + b;
    int d = a - b;

    *ret = common::CINNValuePack{{common::CINNValue(c), common::CINNValue(d)}};
  };

  PackedFunc func(body);

  common::CINNValuePack ret = func(1, 2);
  int c                     = ret[0];
  int d                     = ret[1];

  EXPECT_EQ(c, 3);
  EXPECT_EQ(d, -1);
}
