struct CasInterval {
  template <typename T>
  CasInterval(T l, T r) : l(l), r(r) {}
  int l, r;

  friend std::ostream& operator<<(std::ostream& os, const CasInterval& i) {
    os << "Interval[" << i.l << ", " << i.r << "]";
    return os;
  }
};
