TEST(Solve, basic) {
  Var i("i", Int(32));
  Expr lhs = Expr(i) * 2;
  Expr rhs = Expr(2) * Expr(200);
  Expr res;
  bool is_positive;
  std::tie(res, is_positive) = Solve(lhs, rhs, i);
  LOG(INFO) << "res: " << res;
  EXPECT_TRUE(is_positive);
  EXPECT_TRUE(res == Expr(200));
}
