   void BindSimplify(py::module* m) {
     m->def(
         "simplify",
         [](const Expr& expr) -> Expr {
           auto copied = optim::IRCopy(expr);
           Simplify(&copied);
           return copied;
         },
         py::arg("expr"));

   }
